1. create a bash script containing this code. i.e sync-brain.sh
```bash
#!/bin/bash
# Auto Git Commit & Push Script
# Runs: git add -> git commit -> git push
# Author: Your Name

# Exit immediately if a command exits with a non-zero status
set -e

# Go to the repository directory (change this to your repo path)
REPO_DIR="/path/to/your/repo"
cd "$REPO_DIR" || { echo "Error: Repository directory not found."; exit 1; }

# Check if this is a git repository
if [ ! -d ".git" ]; then
    echo "Error: Not a git repository."
    exit 1
fi

# Stage all changes
git add .

# Check if there are changes to commit
if git diff --cached --quiet; then
    echo "No changes to commit."
    exit 0
fi

# Commit with timestamp
COMMIT_MSG="Auto commit on $(date '+%Y-%m-%d %H:%M:%S')"
git commit -m "$COMMIT_MSG"

# Push to the default remote branch
if ! git push; then
    echo "Error: Failed to push changes."
    exit 1
fi

echo "✅ Changes committed and pushed successfully at $(date)."
```

2. set permission for this script to executable 
`chmod +x /path/to/sync-brain.sh`

3. schedule with cron tab #bash_scripts
* edit crontab with `crontab -e`
* add the following `0 * * * * /bin/bash /path/to/auto_git_push.sh >> /path/to/auto_git_push.log 2>&1`
